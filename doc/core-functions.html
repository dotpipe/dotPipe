domContentLoad(again = false)

Initializes the library by processing various custom tags and setting up event listeners.


function domContentLoad(again = false) {
    doc_set = document.getElementsByTagName("pipe");
    if (again == false) {
        Array.from(doc_set).forEach(function (elem) {
            if (elem.classList.contains("pipe-active"))
                return;
            elem.classList.toggle("pipe-active")
            pipes(elem);
        });
    }
    // ... (rest of the function)
}
